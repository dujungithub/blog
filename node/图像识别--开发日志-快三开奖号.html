8 canvas 二值化处理 getImageData 处理 context.putImageData(imgData, 0, 0);

    imgData = context.getImageData(0, 0, imgArea.width, imgArea.height);
        var index = 255/2;//阈值
        for (var i = 0; i < imgData.data.length; i += 4) {
            var R = imgData.data[i]; //R(0-255)
            var G = imgData.data[i + 1]; //G(0-255)
            var B = imgData.data[i + 2]; //B(0-255)
            var Alpha = imgData.data[i + 3]; //Alpha(0-255)
            var sum = (R + G + B) / 3;
            if (sum > index) {
                imgData.data[i] = 255;
                imgData.data[i + 1] = 255;
                imgData.data[i + 2] = 255;
                imgData.data[i + 3] = Alpha;
            } else {
                imgData.data[i] = 0;
                imgData.data[i + 1] = 0;
                imgData.data[i + 2] = 0;
                imgData.data[i + 3] = Alpha;
            }
        }
        context.putImageData(imgData, 0, 0);

12 资源文件循环读取，需有序，采用bat方式进行批处理重命名 rename.bat

@echo off
setlocal enabledelayedexpansion
set count=0
for /f %%i in ('dir /b *.jpg') do (
		set /a count+=1
		echo 改名：%%i !count!
		rename %%i !count!.jpg
)
