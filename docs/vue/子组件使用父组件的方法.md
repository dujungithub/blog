# 子组件使用父组件的方法：

### 父组件：

```js
<orderItem
    :item="item"
    :oList="oList"
    @payOrderParent="payOrder"
></orderItem>
import orderItem from "../../components/orderItem.vue";
export default {
  name: "My",
  components: {
    orderItem
  },
  methods: {
  	payOrder:function(orderNo){
      let _this = this;
      _this.$router.push({ path: "/OrderPaid/" ,query:{orderNo:orderNo}});
    },//0
  }
}
```

### 子组件

```js
<template>
	<span class="cp-zf" @click="payOrderChild(item.orderNo)">支付运费</span>
</template>
<style scoped>	
	.dsh .cp-zf{color: #333333;border: 0.02rem solid #e3e3e3;}
</style>
<script>
export default {
  name: "orderItem",
  props: {
    item: {
      type: Object
    },
    oList: {
      type: Object
    },
  },
  methods: {
	payOrderChild:function (orderNo) {
      this.$emit('payOrderParent', orderNo);
    },
  }
}
```
